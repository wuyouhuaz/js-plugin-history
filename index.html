数据操作

// 添加一条历史记录

remove('a') 
// 删除一条历史记录

clear() 
// 清空所有历史记录(可以自己添加)

onInputClicked() 
// 显示历史记录框

onInputBlur() 
// 隐藏历史记录框

render() 
// 渲染历史记录框
